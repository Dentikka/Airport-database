USE LIBRARY
set DATEFORMAT dmy

DROP TABLE IF EXISTS BOOK_CREATORS;
DROP TABLE IF EXISTS BOOK_EDITORS;
DROP TABLE IF EXISTS BOOK_TOPIC;
DROP TABLE IF EXISTS CREATOR;
DROP TABLE IF EXISTS PROFESSION;
DROP TABLE IF EXISTS TOPIC;
DROP TABLE IF EXISTS DEAL;

CREATE TABLE READER (
	reader_id SMALLINT IDENTITY PRIMARY KEY,
	first_name VARCHAR(15),
	last_name VARCHAR(15),
	patronym VARCHAR(15),
	email VARCHAR(30),
	phone_number CHAR(12),
	passport CHAR(10) );

CREATE TABLE DEAL (
	deal_id SMALLINT IDENTITY PRIMARY KEY,
	reader_id SMALLINT,
	book_id SMALLINT,
	lending_date DATE DEFAULT GETDATE(),
	returning_date DATE,
	day_x_date AS DATEADD(day, 14, lending_date) );


CREATE TABLE BOOK (
	book_id SMALLINT IDENTITY PRIMARY KEY,
	book_name_id SMALLINT,
	edition_id SMALLINT,
	location_id SMALLINT,
	cost FLOAT,
	date_of_purchase DATE );

CREATE TABLE LOCATION (
	location_id SMALLINT IDENTITY PRIMARY KEY,
	room_number SMALLINT,
	rack_number SMALLINT,
	shelf_number SMALLINT );

CREATE TABLE EDITION (
	edition_id SMALLINT IDENTITY PRIMARY KEY,
	series_id SMALLINT,
	publishing_company_id SMALLINT,
	type_id SMALLINT,
	book_title_id SMALLINT,
	volume_number TINYINT DEFAULT 1,
	quantity SMALLINT,
	periodicity SMALLINT,
	publishing_date DATE );

CREATE TABLE SERIES (
	series_id SMALLINT IDENTITY PRIMARY KEY,
	name VARCHAR(100) );

CREATE TABLE PUBLISHING_COMPANY (
	publishing_company_id SMALLINT IDENTITY PRIMARY KEY,
	name VARCHAR(30) );

CREATE TABLE TYPE (
	type_id SMALLINT IDENTITY PRIMARY KEY,
	name VARCHAR(30) );

CREATE TABLE BOOK_TITLE (
	book_title_id SMALLINT IDENTITY PRIMARY KEY,
	name VARCHAR(500),
	language_id TINYINT,
	original_book_id SMALLINT);

CREATE TABLE ORIGINAL_BOOK (
	original_book_id SMALLINT IDENTITY PRIMARY KEY,
	language_id TINYINT );

CREATE TABLE PROFESSION (
	profession_id SMALLINT IDENTITY PRIMARY KEY,
	name VARCHAR(15) );

CREATE TABLE BOOK_EDITORS (
	profession_id SMALLINT,
	edition_id SMALLINT,
	editor_id SMALLINT );

CREATE TABLE BOOK_CREATORS (
	profession_id SMALLINT,
	original_book_id SMALLINT,
	creator_id SMALLINT );

CREATE TABLE CREATOR (
	creator_id SMALLINT IDENTITY PRIMARY KEY,
	name VARCHAR(100) );

CREATE TABLE BOOK_TOPIC (
	original_book_id SMALLINT,
	topic_id SMALLINT );

CREATE TABLE TOPIC (
	topic_id SMALLINT IDENTITY PRIMARY KEY,
	name VARCHAR(30),
	overtopic_id SMALLINT );

CREATE TABLE LANGUAGE (
	language_id TINYINT IDENTITY PRIMARY KEY,
	language VARCHAR(15) );

ALTER TABLE DEAL
	ADD CONSTRAINT FK_Deal_Reader FOREIGN KEY (reader_id)
	REFERENCES READER (reader_id)
	ON DELETE CASCADE
	ON UPDATE CASCADE
;

ALTER TABLE DEAL
	ADD CONSTRAINT FK_Deal_Book FOREIGN KEY (book_id)
	REFERENCES BOOK (book_id)
	ON DELETE CASCADE
	ON UPDATE CASCADE
;

   ADD CONSTRAINT FK_Book_Location FOREIGN KEY (location_id)
      REFERENCES LOCATION (location_id)
      ON DELETE CASCADE
      ON UPDATE CASCADE
;

   ADD CONSTRAINT FK_Book_Edition FOREIGN KEY (edition_id)

ALTER TABLE EDITION
